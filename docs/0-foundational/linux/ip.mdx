# retrive just data of specif interface

# retrive just a lit of physical interfaces
find /sys/class/net -type l -not -lname '*virtual*' -printf '%f\n' | xargs -I '{}'  ip -h -4 addr show {}

## from: man find
### -printf 
#### %f     Print the basename; the file's name with any leading directories removed (only the last element).  For /, the result is `/'.  See the EXAMPLES section for an example.
